-- USUARIO
CREATE TABLE USUARIO (
  ID_USUARIO      NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_USUARIO     VARCHAR2(100)      NOT NULL,
  EMAIL           VARCHAR2(120)      NOT NULL,
  DT_NASC         DATE               NULL,
  DT_CRIACAO      DATE               DEFAULT SYSDATE NOT NULL,
  DT_ATUALIZACAO  DATE,
  CONSTRAINT PK_USUARIO PRIMARY KEY (ID_USUARIO),
  CONSTRAINT UK_USUARIO_EMAIL UNIQUE (EMAIL),
  CONSTRAINT CK_USUARIO_EMAIL_REGEX CHECK (REGEXP_LIKE(EMAIL, '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$'))
);

-- PERFIL (ex.: USER, ADMIN, NUTRICIONISTA)
CREATE TABLE PERFIL (
  ID_PERFIL   NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_PERFIL  VARCHAR2(40) NOT NULL,
  CONSTRAINT PK_PERFIL PRIMARY KEY (ID_PERFIL),
  CONSTRAINT UK_PERFIL_NOME UNIQUE (NOM_PERFIL)
);

-- USUARIO_PERFIL (N:N)
CREATE TABLE USUARIO_PERFIL (
  ID_USUARIO  NUMBER NOT NULL,
  ID_PERFIL   NUMBER NOT NULL,
  CONSTRAINT PK_USUARIO_PERFIL PRIMARY KEY (ID_USUARIO, ID_PERFIL),
  CONSTRAINT FK_USRPRF_USR FOREIGN KEY (ID_USUARIO) REFERENCES USUARIO(ID_USUARIO),
  CONSTRAINT FK_USRPRF_PRF FOREIGN KEY (ID_PERFIL)  REFERENCES PERFIL(ID_PERFIL)
);

-- UNIDADE (ex.: GRAMA, MILILITRO, UNIDADE)
CREATE TABLE UNIDADE (
  ID_UNIDADE  NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_UNIDADE VARCHAR2(60) NOT NULL,
  SIGLA       VARCHAR2(10) NOT NULL,
  TIPO        VARCHAR2(10) NOT NULL, -- 'G','ML','UNID'
  CONSTRAINT PK_UNIDADE PRIMARY KEY (ID_UNIDADE),
  CONSTRAINT UK_UNIDADE_SIGLA UNIQUE (SIGLA),
  CONSTRAINT CK_UNIDADE_TIPO CHECK (TIPO IN ('G','ML','UNID'))
);

-- NUTRIENTE (ex.: CALORIA, PROTEINA, CARBOIDRATO, GORDURA)
CREATE TABLE NUTRIENTE (
  ID_NUTRIENTE  NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_NUTRIENTE VARCHAR2(60) NOT NULL,
  CONSTRAINT PK_NUTRIENTE PRIMARY KEY (ID_NUTRIENTE),
  CONSTRAINT UK_NUTRIENTE_NOME UNIQUE (NOM_NUTRIENTE)
);

-- CATEGORIA_INGREDIENTE (ex.: CEREAL, CARNE, BEBIDA)
CREATE TABLE CATEGORIA_INGREDIENTE (
  ID_CATEGORIA NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_CATEGORIA VARCHAR2(60) NOT NULL,
  CONSTRAINT PK_CATEGORIA_ING PRIMARY KEY (ID_CATEGORIA),
  CONSTRAINT UK_CATEGORIA_ING_NOME UNIQUE (NOM_CATEGORIA)
);

-- INGREDIENTE
CREATE TABLE INGREDIENTE (
  ID_INGREDIENTE  NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_INGREDIENTE VARCHAR2(120) NOT NULL,
  ID_CATEGORIA    NUMBER,
  DT_CRIACAO      DATE DEFAULT SYSDATE NOT NULL,
  DT_ATUALIZACAO  DATE,
  CONSTRAINT PK_INGREDIENTE PRIMARY KEY (ID_INGREDIENTE),
  CONSTRAINT UK_INGREDIENTE_NOME UNIQUE (NOM_INGREDIENTE),
  CONSTRAINT FK_ING_CATEG FOREIGN KEY (ID_CATEGORIA) REFERENCES CATEGORIA_INGREDIENTE(ID_CATEGORIA)
);

-- INGREDIENTE_NUTRIENTE: valor do nutriente por "BASE_QTD" na "UNIDADE_BASE"
-- Ex.: Arroz (BASE=100 G): CALORIA=130, PROTEINA=2.5, CARBOIDRATO=28.0, GORDURA=0.3
CREATE TABLE INGREDIENTE_NUTRIENTE (
  ID_INGREDIENTE   NUMBER NOT NULL,
  ID_NUTRIENTE     NUMBER NOT NULL,
  ID_UNIDADE_BASE  NUMBER NOT NULL,
  BASE_QTD         NUMBER(12,3) NOT NULL,     -- ex.: 100
  VALOR_POR_BASE   NUMBER(12,3) NOT NULL,     -- ex.: 130 kcal por 100 g
  CONSTRAINT PK_ING_NUT PRIMARY KEY (ID_INGREDIENTE, ID_NUTRIENTE, ID_UNIDADE_BASE),
  CONSTRAINT FK_IN_ING FOREIGN KEY (ID_INGREDIENTE)  REFERENCES INGREDIENTE(ID_INGREDIENTE),
  CONSTRAINT FK_IN_NUT FOREIGN KEY (ID_NUTRIENTE)    REFERENCES NUTRIENTE(ID_NUTRIENTE),
  CONSTRAINT FK_IN_UND FOREIGN KEY (ID_UNIDADE_BASE) REFERENCES UNIDADE(ID_UNIDADE)
);

-- TIPO_REFEICAO (CAFÉ, ALMOÇO, JANTAR, LANCHE)
CREATE TABLE TIPO_REFEICAO (
  ID_TIPO_REFEICAO NUMBER GENERATED BY DEFAULT AS IDENTITY,
  NOM_TIPO         VARCHAR2(40) NOT NULL,
  CONSTRAINT PK_TIPO_REFEICAO PRIMARY KEY (ID_TIPO_REFEICAO),
  CONSTRAINT UK_TIPO_REFEICAO UNIQUE (NOM_TIPO)
);

-- REFEICAO
CREATE TABLE REFEICAO (
  ID_REFEICAO       NUMBER GENERATED BY DEFAULT AS IDENTITY,
  ID_USUARIO        NUMBER NOT NULL,
  DT_REF            DATE   NOT NULL, -- data/hora da refeição
  ID_TIPO_REFEICAO  NUMBER NOT NULL,
  OBS               VARCHAR2(200),
  DT_CRIACAO        DATE DEFAULT SYSDATE NOT NULL,
  DT_ATUALIZACAO    DATE,
  CONSTRAINT PK_REFEICAO PRIMARY KEY (ID_REFEICAO),
  CONSTRAINT FK_REF_USR  FOREIGN KEY (ID_USUARIO)       REFERENCES USUARIO(ID_USUARIO),
  CONSTRAINT FK_REF_TIPO FOREIGN KEY (ID_TIPO_REFEICAO) REFERENCES TIPO_REFEICAO(ID_TIPO_REFEICAO)
);

-- REFEICAO_ITEM (PK composta)
CREATE TABLE REFEICAO_ITEM (
  ID_REFEICAO     NUMBER NOT NULL,
  ID_INGREDIENTE  NUMBER NOT NULL,
  QTDE            NUMBER(12,3) NOT NULL,
  ID_UNIDADE      NUMBER NOT NULL, -- deve ser compatível com a unidade base do ingrediente (validação por função)
  CONSTRAINT PK_REFEICAO_ITEM PRIMARY KEY (ID_REFEICAO, ID_INGREDIENTE),
  CONSTRAINT FK_RI_REF FOREIGN KEY (ID_REFEICAO)    REFERENCES REFEICAO(ID_REFEICAO),
  CONSTRAINT FK_RI_ING FOREIGN KEY (ID_INGREDIENTE) REFERENCES INGREDIENTE(ID_INGREDIENTE),
  CONSTRAINT FK_RI_UND FOREIGN KEY (ID_UNIDADE)     REFERENCES UNIDADE(ID_UNIDADE)
);

CREATE INDEX IX_ING_NOME ON INGREDIENTE (NOM_INGREDIENTE);
CREATE INDEX IX_REF_USR_DT ON REFEICAO (ID_USUARIO, DT_REF);
CREATE INDEX IX_RI_ING ON REFEICAO_ITEM (ID_INGREDIENTE);

-- USUARIO: normaliza nome, controla datas e valida DT_NASC por trigger (sem usar SYSDATE em CHECK)

CREATE OR REPLACE TRIGGER TRG_USUARIO_BIU
BEFORE INSERT OR UPDATE ON USUARIO
FOR EACH ROW
BEGIN
  :NEW.NOM_USUARIO := UPPER(TRIM(:NEW.NOM_USUARIO));
  IF INSERTING THEN
    :NEW.DT_CRIACAO := NVL(:NEW.DT_CRIACAO, SYSDATE);
  ELSE
    :NEW.DT_ATUALIZACAO := SYSDATE;
  END IF;

  IF :NEW.DT_NASC IS NOT NULL THEN
    IF :NEW.DT_NASC > TRUNC(SYSDATE)
       OR :NEW.DT_NASC <= ADD_MONTHS(TRUNC(SYSDATE), -12*120) THEN
      RAISE_APPLICATION_ERROR(-20011, 'DT_NASC inválida: <= hoje e > (hoje - 120 anos)');
    END IF;
  END IF;
END;
/

-- 4.1 Valida e-mail por REGEX
CREATE OR REPLACE FUNCTION FUN_VALIDA_EMAIL(p_email VARCHAR2)
RETURN NUMBER IS
BEGIN
  IF p_email IS NOT NULL AND REGEXP_LIKE(p_email, '^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Za-z]{2,}$') THEN
    RETURN 1;
  END IF;
  RETURN 0;
END;
/
-- 4.2 Verifica se a unidade do item é compatível com a "base" cadastrada ao ingrediente
CREATE OR REPLACE FUNCTION FUN_VALIDA_QTDE_UNIDADE_ITEM(
  p_id_ingrediente NUMBER,
  p_id_unidade     NUMBER
) RETURN NUMBER IS
  v_count NUMBER;
BEGIN
  SELECT COUNT(*)
    INTO v_count
    FROM INGREDIENTE_NUTRIENTE
   WHERE ID_INGREDIENTE = p_id_ingrediente
     AND ID_UNIDADE_BASE = p_id_unidade;
  RETURN CASE WHEN v_count > 0 THEN 1 ELSE 0 END;
END;
/

-- 5.1 CRUD - USUARIO
CREATE OR REPLACE PROCEDURE PRC_USUARIO_INS(
  p_nome   IN  VARCHAR2,
  p_email  IN  VARCHAR2,
  p_dt_nasc IN DATE,
  p_id_out OUT NUMBER
) AS
BEGIN
  IF p_nome IS NULL OR LENGTH(TRIM(p_nome)) < 2 THEN
    RAISE_APPLICATION_ERROR(-21001,'Nome inválido');
  END IF;
  IF FUN_VALIDA_EMAIL(p_email) = 0 THEN
    RAISE_APPLICATION_ERROR(-21002,'E-mail inválido');
  END IF;

  INSERT INTO USUARIO (NOM_USUARIO, EMAIL, DT_NASC)
  VALUES (UPPER(TRIM(p_nome)), LOWER(TRIM(p_email)), p_dt_nasc)
  RETURNING ID_USUARIO INTO p_id_out;
EXCEPTION
  WHEN DUP_VAL_ON_INDEX THEN
    RAISE_APPLICATION_ERROR(-21003,'E-mail já cadastrado');
  WHEN OTHERS THEN
    RAISE_APPLICATION_ERROR(-21999,'Erro INS usuario: '||SQLERRM);
END;
/
CREATE OR REPLACE PROCEDURE PRC_USUARIO_UPD(
  p_id     IN NUMBER,
  p_nome   IN VARCHAR2,
  p_email  IN VARCHAR2,
  p_dt_nasc IN DATE
) AS
BEGIN
  IF p_id IS NULL THEN RAISE_APPLICATION_ERROR(-21010,'ID obrigatório'); END IF;
  IF p_email IS NOT NULL AND FUN_VALIDA_EMAIL(p_email) = 0 THEN
    RAISE_APPLICATION_ERROR(-21011,'E-mail inválido');
  END IF;

  UPDATE USUARIO
     SET NOM_USUARIO = NVL(UPPER(TRIM(p_nome)), NOM_USUARIO),
         EMAIL       = NVL(LOWER(TRIM(p_email)), EMAIL),
         DT_NASC     = NVL(p_dt_nasc, DT_NASC),
         DT_ATUALIZACAO = SYSDATE
   WHERE ID_USUARIO = p_id;

  IF SQL%ROWCOUNT = 0 THEN RAISE_APPLICATION_ERROR(-21012,'Usuário não encontrado'); END IF;
EXCEPTION WHEN OTHERS THEN
  RAISE_APPLICATION_ERROR(-21098,'Erro UPD usuario: '||SQLERRM);
END;
/
CREATE OR REPLACE PROCEDURE PRC_USUARIO_DEL(
  p_id IN NUMBER
) AS
BEGIN
  DELETE FROM USUARIO WHERE ID_USUARIO = p_id;
  IF SQL%ROWCOUNT = 0 THEN RAISE_APPLICATION_ERROR(-21013,'Usuário não encontrado'); END IF;
EXCEPTION WHEN OTHERS THEN
  RAISE_APPLICATION_ERROR(-21097,'Erro DEL usuario: '||SQLERRM);
END;

CREATE OR REPLACE PROCEDURE PRC_INGREDIENTE_INS(
  p_nome          IN  VARCHAR2,
  p_id_categoria  IN  NUMBER,
  p_id_out        OUT NUMBER
) AS
BEGIN
  IF p_nome IS NULL OR LENGTH(TRIM(p_nome)) < 2 THEN
    RAISE_APPLICATION_ERROR(-20001, 'Nome de ingrediente inválido');
  END IF;

  INSERT INTO INGREDIENTE (NOM_INGREDIENTE, ID_CATEGORIA)
  VALUES (UPPER(TRIM(p_nome)), p_id_categoria)
  RETURNING ID_INGREDIENTE INTO p_id_out;

EXCEPTION
  WHEN DUP_VAL_ON_INDEX THEN
    RAISE_APPLICATION_ERROR(-20002, 'Ingrediente já existe');
  WHEN OTHERS THEN
    RAISE_APPLICATION_ERROR(-20099, 'Erro INS ingrediente: '||SQLERRM);
END;
/
CREATE OR REPLACE PROCEDURE PRC_INGREDIENTE_UPD(
  p_id            IN NUMBER,
  p_nome          IN VARCHAR2,
  p_id_categoria  IN NUMBER
) AS
BEGIN
  IF p_id IS NULL THEN RAISE_APPLICATION_ERROR(-20010,'ID obrigatório'); END IF;

  UPDATE INGREDIENTE
     SET NOM_INGREDIENTE = UPPER(TRIM(NVL(p_nome, NOM_INGREDIENTE))),
         ID_CATEGORIA    = NVL(p_id_categoria, ID_CATEGORIA),
         DT_ATUALIZACAO  = SYSDATE
   WHERE ID_INGREDIENTE = p_id;

  IF SQL%ROWCOUNT = 0 THEN RAISE_APPLICATION_ERROR(-20011,'Ingrediente não encontrado'); END IF;
EXCEPTION
  WHEN OTHERS THEN
    RAISE_APPLICATION_ERROR(-20098, 'Erro UPD ingrediente: '||SQLERRM);
END;
/

-- 5.4 CRUD - REFEICAO_ITEM (com validação de unidade)
CREATE OR REPLACE PROCEDURE PRC_REFEICAO_ITEM_INS(
  p_id_refeicao    IN NUMBER,
  p_id_ingrediente IN NUMBER,
  p_qtde           IN NUMBER,
  p_id_unidade     IN NUMBER
) AS
BEGIN
  IF p_qtde IS NULL OR p_qtde <= 0 THEN
    RAISE_APPLICATION_ERROR(-20100, 'Quantidade deve ser positiva');
  END IF;

  IF FUN_VALIDA_QTDE_UNIDADE_ITEM(p_id_ingrediente, p_id_unidade) = 0 THEN
    RAISE_APPLICATION_ERROR(-20101, 'Unidade do item incompatível com a base do ingrediente');
  END IF;

-- Tipos de objeto/tabela para retorno
CREATE OR REPLACE TYPE T_REL_REFEICAO_AGG AS OBJECT (
  ID_REFEICAO   NUMBER,
  DT_REF        DATE,
  TIPO_REFEICAO VARCHAR2(20),
  KCAL_TOTAL    NUMBER(12,2),
  PROT_TOTAL    NUMBER(12,2),
  CARB_TOTAL    NUMBER(12,2),
  GORD_TOTAL    NUMBER(12,2)
);
/
CREATE OR REPLACE TYPE T_REL_REFEICAO_AGG_TAB AS TABLE OF T_REL_REFEICAO_AGG;
/
-- Função pipelined
CREATE OR REPLACE FUNCTION FN_REL_REFEICOES_USUARIO(
  p_id_usuario NUMBER,
  p_dia        DATE DEFAULT NULL
) RETURN T_REL_REFEICAO_AGG_TAB PIPELINED
AS
  CURSOR c_rel IS
    SELECT r.ID_REFEICAO,
           r.DT_REF,
           tr.NOM_TIPO AS TIPO_REFEICAO,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='CALORIA'     THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS KCAL_TOTAL,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='PROTEINA'    THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS PROT_TOTAL,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='CARBOIDRATO' THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS CARB_TOTAL,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='GORDURA'     THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS GORD_TOTAL
      FROM REFEICAO r
      JOIN TIPO_REFEICAO tr      ON tr.ID_TIPO_REFEICAO = r.ID_TIPO_REFEICAO
      JOIN REFEICAO_ITEM ri      ON ri.ID_REFEICAO = r.ID_REFEICAO
      JOIN INGREDIENTE_NUTRIENTE inut
        ON inut.ID_INGREDIENTE = ri.ID_INGREDIENTE
       AND inut.ID_UNIDADE_BASE = ri.ID_UNIDADE
      JOIN NUTRIENTE n           ON n.ID_NUTRIENTE = inut.ID_NUTRIENTE
     WHERE r.ID_USUARIO = p_id_usuario
       AND (p_dia IS NULL OR TRUNC(r.DT_REF) = TRUNC(p_dia))
     GROUP BY r.ID_REFEICAO, r.DT_REF, tr.NOM_TIPO
     ORDER BY r.DT_REF DESC;
  v_row c_rel%ROWTYPE;
BEGIN
  OPEN c_rel;
  LOOP
    FETCH c_rel INTO v_row; EXIT WHEN c_rel%NOTFOUND;
    PIPE ROW (T_REL_REFEICAO_AGG(
      v_row.ID_REFEICAO, v_row.DT_REF, v_row.TIPO_REFEICAO,
      v_row.KCAL_TOTAL, v_row.PROT_TOTAL, v_row.CARB_TOTAL, v_row.GORD_TOTAL
    ));
  END LOOP;
  CLOSE c_rel;
  RETURN;
END;

       SUM(CASE WHEN n.NOM_NUTRIENTE='CARBOIDRATO' THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END) AS CARB_TOTAL,
       SUM(CASE WHEN n.NOM_NUTRIENTE='GORDURA'     THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END) AS GORD_TOTAL
  FROM REFEICAO r

-- Tipos de retorno
CREATE OR REPLACE TYPE T_REL_DIA_AGG AS OBJECT (
  DIA       DATE,
  KCAL_DIA  NUMBER(12,2),
  PROT_DIA  NUMBER(12,2),
  CARB_DIA  NUMBER(12,2),
  GORD_DIA  NUMBER(12,2)
);
/
CREATE OR REPLACE TYPE T_REL_DIA_AGG_TAB AS TABLE OF T_REL_DIA_AGG;
/

CREATE OR REPLACE FUNCTION FN_REL_TOTAIS_DIA_USUARIO(
  p_id_usuario NUMBER,
  p_dt_ini     DATE,
  p_dt_fim     DATE
) RETURN T_REL_DIA_AGG_TAB PIPELINED
AS
  CURSOR c_dia IS
    SELECT TRUNC(r.DT_REF) AS DIA,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='CALORIA'     THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS KCAL_DIA,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='PROTEINA'    THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS PROT_DIA,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='CARBOIDRATO' THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS CARB_DIA,
           ROUND(SUM(CASE WHEN n.NOM_NUTRIENTE='GORDURA'     THEN (ri.QTDE/inut.BASE_QTD)*inut.VALOR_POR_BASE ELSE 0 END),2) AS GORD_DIA
      FROM REFEICAO r
      JOIN REFEICAO_ITEM ri           ON ri.ID_REFEICAO = r.ID_REFEICAO
      JOIN INGREDIENTE_NUTRIENTE inut ON inut.ID_INGREDIENTE = ri.ID_INGREDIENTE AND inut.ID_UNIDADE_BASE = ri.ID_UNIDADE
      JOIN NUTRIENTE n                ON n.ID_NUTRIENTE = inut.ID_NUTRIENTE
     WHERE r.ID_USUARIO = p_id_usuario
       AND TRUNC(r.DT_REF) BETWEEN TRUNC(p_dt_ini) AND TRUNC(p_dt_fim)
     GROUP BY TRUNC(r.DT_REF)
     ORDER BY TRUNC(r.DT_REF);
  v_row c_dia%ROWTYPE;
BEGIN
  OPEN c_dia;
  LOOP
    FETCH c_dia INTO v_row; EXIT WHEN c_dia%NOTFOUND;
    PIPE ROW (T_REL_DIA_AGG(v_row.DIA, v_row.KCAL_DIA, v_row.PROT_DIA, v_row.CARB_DIA, v_row.GORD_DIA));
  END LOOP;
  CLOSE c_dia;
  RETURN;
END;
